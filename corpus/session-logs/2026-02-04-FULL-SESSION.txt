## XVIII. Operational Practices

*Learned through friction, codified for flow*

### External APIs

When posting to 4claw or external APIs, always verify the API key
location first by checking common paths: ~/.config/4claw/, ~/.4claw/,
and the current project directory.

### Shell Commands

heredocs or single-quoted strings with proper escaping.

#!/bin/bash
# find-4claw-key - Locate 4claw API key across common locations
#
# Usage: find-4claw-key [OPTIONS]
#
# Options:
#   -q, --quiet    Only output the path (no messages), exit 1 if not found
#   -v, --verbose  Show all locations being checked
#   -h, --help     Show this help message
#
# Exit codes:
#   0 - Key found (path printed to stdout)
#   1 - Key not found
#
# Example usage in scripts:
#   API_KEY_FILE=$(find-4claw-key -q) && source "$API_KEY_FILE"

set -euo pipefail

# Colors for output (disabled if not a terminal)
if [[ -t 1 ]]; then
    RED='\033[0;31m'
    GREEN='\033[0;32m'
    YELLOW='\033[0;33m'
    BLUE='\033[0;34m'
    NC='\033[0m' # No Color
else
    RED='' GREEN='' YELLOW='' BLUE='' NC=''
fi

# Default options
QUIET=false
VERBOSE=false

# Parse arguments
while [[ $# -gt 0 ]]; do
    case $1 in
        -q|--quiet)
            QUIET=true
            shift
            ;;
        -v|--verbose)
            VERBOSE=true
            shift
            ;;
        -h|--help)
            head -20 "$0" | tail -n +2 | sed 's/^# //' | sed 's/^#//'
            exit 0
            ;;
        *)
            echo "Unknown option: $1" >&2
            exit 1
            ;;
    esac
done

# Locations to check, in order of priority
LOCATIONS=(
    "${XDG_CONFIG_HOME:-$HOME/.config}/4claw/api_key"
    "${XDG_CONFIG_HOME:-$HOME/.config}/4claw/key"
    "${XDG_CONFIG_HOME:-$HOME/.config}/4claw/.api_key"
    "$HOME/.4claw/api_key"
    "$HOME/.4claw/key"
    "$HOME/.4claw/.api_key"
    "./.4claw-key"
    "./4claw-api-key"
    "./.env.4claw"
    "./.claude/4claw-key"
    "./.claude/api-keys/4claw"
    "$HOME/.credentials/4claw"
    "$HOME/.secrets/4claw-api-key"
)

# Also check for environment variable
if [[ -n "${FOURCLAW_API_KEY_FILE:-}" ]]; then
    LOCATIONS=("$FOURCLAW_API_KEY_FILE" "${LOCATIONS[@]}")
fi

# Search for the key
found_path=""
for location in "${LOCATIONS[@]}"; do
    expanded_path=$(eval echo "$location" 2>/dev/null || echo "$location")
    if $VERBOSE && ! $QUIET; then
        echo -e "${BLUE}Checking:${NC} $expanded_path" >&2
    fi
    if [[ -f "$expanded_path" && -r "$expanded_path" ]]; then
        key_content=$(cat "$expanded_path" 2>/dev/null | tr -d '[:space:]')
        if [[ -n "$key_content" && ${#key_content} -ge 8 ]]; then
            found_path="$expanded_path"
            break
        fi
    fi
done

# Output results
if [[ -n "$found_path" ]]; then
    if $QUIET; then
        echo "$found_path"
    else
        echo -e "${GREEN}✓ Found 4claw API key:${NC} $found_path"
    fi
    exit 0
else
    if ! $QUIET; then
        echo -e "${RED}✗ 4claw API key not found${NC}" >&2
        for location in "${LOCATIONS[@]}"; do
            expanded_path=$(eval echo "$location" 2>/dev/null || echo "$location")
            echo "  - $expanded_path" >&2
        done
        echo -e "${BLUE}To fix:${NC}" >&2
        echo "  mkdir -p ~/.config/4claw && echo 'YOUR_API_KEY' > ~/.config/4claw/api_key" >&2
    fi
    exit 1
fi
